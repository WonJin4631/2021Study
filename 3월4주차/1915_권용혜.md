# 문제: 가장 큰 정사각형

```cpp
#include <iostream>
#include <algorithm>
#include <climits>
#include <string>
#include <stack>
#include <queue>
#include <deque>
using namespace std;

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

    int arr[1001][1001]={0,};
    int dp[1001][1001]={0,}; // arr[i][j]가 right bottom 꼭지점인 1로 가득찬 정사각형의 변의 최대 길이 
    int n;
    int m;
    int answer=0;
    
    cin >> n >> m;
    for(int i = 1; i <=n;i++){
        string s;
        cin>>s;
        for(int j = 0 ;j < m ; j++){
            arr[i][j+1] = s[j]-48;
            dp[i][j+1] = s[j]-48;
            if(dp[i][j+1]) answer=1;
        }
    }
    
    
    for(int i=1; i<=n; i++)
        for(int j=1; j<=m; j++)
        {
            if(arr[i-1][j-1] == 1 
            and arr[i-1][j] == 1 
            and arr[i][j-1] == 1)
            {
                dp[i][j] = min(dp[i-1][j-1], min(dp[i-1][j], dp[i][j-1])) + 1;
                answer = max(answer, dp[i][j] * dp[i][j]);
            }
        }
    cout << answer;
    return 0;
}
```
